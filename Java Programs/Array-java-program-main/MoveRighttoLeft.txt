package arrays;
import java.util.Arrays;
public class MoveZeroRight {

	public static void main(String[] args) {
		int a[] = {4,0,1,0,5};
		int temp;
		for(int i=0;i<a.length-1;i++) {
			if(a[i]==0) {
				for(int j=a.length-1;j>0;j++) {
					if(a[j]!=0) {
					temp=a[i];
                     a[i]=a[j];
				    a[j]=temp;
					
		}
				}
		}
		}
		System.out.println(Arrays.toString(a));
}
}
